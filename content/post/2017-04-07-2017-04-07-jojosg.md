// precondition: m5c5
 class Key
 {
 // data
 private:
 Writer dreams;
 public:
 Composer nightmares;

// method
 string Get()
 {
 Track t = findTrack(nightmares.Last() + " & " + dreams.Last());
 return t.Prev().Prev().Prev().Prev().Title();
 }
 }
 // postcondition: n7zy

代码描述出前缀为m5c5，后缀为n7zy

另外的提示是：

private:Writer dreams

public:Composer nightmares
